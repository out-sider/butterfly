ef _token_type_name(token_type):
   NoneType = type(None)
   if token_type is NoneType:
       return 'end of file'
   elif token_type is KeywordToken:
       return 'keyword'
   elif token_type is NumberToken:
       return 'number'
   else:
       return 'unknown'
